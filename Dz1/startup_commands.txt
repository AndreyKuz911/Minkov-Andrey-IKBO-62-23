# Вывод содержимого текущей директории

# Переход в поддиректорию 'subdir1'

# Вывод содержимого 'subdir1'
ls

ls

# Вывод последних 10 строк файла 'file1_moved.txt'
tail file1_moved.txt

# Возврат в родительскую директорию
cd ..

# Повторный вывод содержимого корневой директории
ls

exit
